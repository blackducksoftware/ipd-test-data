def display_grid(rows, cols):
    total_cells = rows * cols
    for i in range(rows):
        row = ""
        for j in range(cols):
            if (i+j) % 2 == 0:
                row += "+"
            else:
                row += "-"
        print(row)
    print("\nThe total number of cells in the grid is", total_cells)

rows = int(input("Enter the number of rows: "))
cols = int(input("Enter the number of columns: ")

display_grid(rows, cols)
