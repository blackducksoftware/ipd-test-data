```python
def fibonacci(n):
    if n <= 1:
        return n
    else:
        return fibonacci(n-1) + fibonacci(n-2)

def print_fibonacci_sequence(n):
    for i in range(n):
        print(fibonacci(i))

# Get user input for the maximum number in the Fibonacci sequence
max_num = int(input("Enter the maximum number in the Fibonacci sequence: "))

print("Fibonacci sequence up to", max_num, ":")
print_fibonacci_sequence(max_num)
```
