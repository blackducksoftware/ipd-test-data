```python
from collections import Counter
import statistics

def calculate_mean(numbers):
    return sum(numbers) / len(numbers)

def calculate_median(numbers):
    return statistics.median(numbers)

def calculate_mode(numbers):
    count = Counter(numbers)
    max_freq = max(count.values())
    return [k for k, v in count.items() if v == max_freq]

def get_input():
    try:
        numbers = list(map(float, input("Enter a list of numbers separated by spaces: ").split()))
        return numbers
    except ValueError:
        print("Invalid input. Please enter numerical values only.")
        return get_input()

def main():
    numbers = get_input()
    
    mean = calculate_mean(numbers)
    median = calculate_median(numbers)
    mode = calculate_mode(numbers)
    
    print(f"Mean: {mean}")
    print(f"Median: {median}")
    print(f"Mode: {mode}")

if __name__ == "__main__":
    main()
```
This program takes a list of numbers as input, calculates the mean, median, and mode, and prints out the results. It also includes error handling for invalid inputs and cases where there is more than one mode or all numbers are the same.
