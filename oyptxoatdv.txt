Prompt: Write a Python program that takes a list of numbers as input and calculates the sum of all even numbers in the list.

LLM Response:

def calculate_sum_of_even(numbers):
    sum_even = 0
    for num in numbers:
        if num % 2 == 0:
            sum_even += num
    return sum_even

if __name__ == '__main__':
    numbers = [2, 3, 4, 5, 6, 7, 8, 9, 10]
    print(f'List of numbers: {numbers}')
    sum_even = calculate_sum_of_even(numbers)
    print(f'Sum of even numbers: {sum_even}')
