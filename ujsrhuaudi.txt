```python
def fibonacci_sequence(n):
    sequence = []
    a, b = 0, 1
    while a < n:
        sequence.append(a)
        a, b = b, a + b
    return sequence

def main():
    try:
        number = int(input("Enter a number to generate the Fibonacci sequence up to: "))
        if number <= 0:
            print("Please enter a positive number.")
        else:
            fib_sequence = fibonacci_sequence(number)
            print("Fibonacci sequence up to", number, ":")
            print(fib_sequence)
    except ValueError:
        print("Invalid input. Please enter a valid number.")

if __name__ == "__main__":
    main()
```
