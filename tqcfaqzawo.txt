Here is the code for the rock-paper-scissors game simulation in Python:

```python
import random

def get_user_choice():
    while True:
        user_choice = input("Enter your choice (rock, paper, or scissors) or 'quit' to exit: ").lower()
        if user_choice in ['rock', 'paper', 'scissors', 'quit']:
            return user_choice
        else:
            print("Invalid choice. Please enter either 'rock', 'paper', 'scissors', or 'quit'.")

def get_computer_choice():
    return random.choice(['rock', 'paper', 'scissors'])

def determine_winner(user_choice, computer_choice):
    if user_choice == 'quit':
        return 'quit'
    if user_choice == computer_choice:
        return 'It\'s a tie!'
    if (user_choice == 'rock' and computer_choice == 'scissors') or (user_choice == 'scissors' and computer_choice == 'paper') or (user_choice == 'paper' and computer_choice == 'rock'):
        return 'You win!'
    else:
        return 'Computer wins!'

def main():
    while True:
        user_choice = get_user_choice()
        computer_choice = get_computer_choice()
        
        if determine_winner(user_choice, computer_choice) == 'quit':
            print('Goodbye!')
            break
        
        print(f"Your choice: {user_choice}")
        print(f"Computer's choice: {computer_choice}")
        print(determine_winner(user_choice, computer_choice))

if __name__ == "__main__":
    main()
```

You can copy and paste this code into a Python environment and run it to play the rock-paper-scissors game. The program will prompt you to enter your choice, generate a random choice for the computer, and determine the winner based on the rules of the game. The game will continue to run until you choose to quit by entering 'quit'. Enjoy playing!
