import random

def generate_deck():
    deck = []
    for _ in range(4):
        for i in range(2, 11):
            deck.append(i)
        for _ in range(3):
            deck.extend([10, 10, 10])
    random.shuffle(deck)
    return deck

def calculate_score(hand):
    score = sum(hand)
    if 11 in hand and score > 21:
        hand.remove(11)
        hand.append(1)
        score = sum(hand)
    return score

def blackjack():
    deck = generate_deck()
    player_hand = [deck.pop(), deck.pop()]
    computer_hand = [deck.pop(), deck.pop()]
    
    print(f"Player's hand: {player_hand}")
    print(f"Computer's hand: [{computer_hand[0]}, X]")
    
    while True:
        decision = input("Do you want to hit or stand? ").lower()
        
        if decision == "hit":
            player_hand.append(deck.pop())
            print(f"Player's hand: {player_hand}")
            player_score = calculate_score(player_hand)
            if player_score > 21:
                print("Bust! Computer wins.")
                break
        elif decision == "stand":
            player_score = calculate_score(player_hand)
            break
        else:
            print("Invalid input. Please try again.")
    
    while calculate_score(computer_hand) < 17:
        computer_hand.append(deck.pop())
    
    computer_score = calculate_score(computer_hand)
    
    print(f"Player's final hand: {player_hand} ({player_score})")
    print(f"Computer's final hand: {computer_hand} ({computer_score})")
    
    if computer_score > 21 or player_score == 21 or (player_score > computer_score and player_score <= 21):
        print("Player wins!")
    else:
        print("Computer wins!")

blackjack()
