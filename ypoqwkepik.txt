```python
def reverse_string(s):
    if len(s) == 0:
        return s
    else:
        return reverse_string(s[1:]) + s[0]

def main():
    user_input = input("Please enter a string: ")
    reversed_string = reverse_string(user_input)
    print("The reversed string is:", reversed_string)

if __name__ == "__main__":
    main()
```
