```python
import csv

def calculate_average(csv_file, column_name):
    total_sum = 0
    total_rows = 0
    
    try:
        with open(csv_file, newline='') as csvfile:
            reader = csv.DictReader(csvfile)
            for row in reader:
                try:
                    total_sum += float(row[column_name])
                    total_rows += 1
                except (KeyError, ValueError):
                    continue
                
        if total_rows == 0:
            return "No valid data found in the specified column"
        
        average = total_sum / total_rows
        return f"The average value of column '{column_name}' is {average}. Total number of rows: {total_rows}"
    
    except FileNotFoundError:
        return f"Error: File '{csv_file}' not found"
    except Exception as e:
        return f"An error occurred: {e}"

# Specify the CSV file and column to calculate average value
csv_file = 'data.csv'
column_name = 'column_name'

print(calculate_average(csv_file, column_name))
```
