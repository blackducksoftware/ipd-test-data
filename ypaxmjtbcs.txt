# Read a list of numbers from the user
numbers = input("Enter a list of numbers separated by spaces: ").split()
numbers = [float(num) for num in numbers]

# Calculate the average of the numbers
average = sum(numbers) / len(numbers)

# Determine which numbers are above the average
above_average = [num for num in numbers if num > average]

# Output the average
print("Average:", average)

# Output the numbers that are above the average
print("Numbers above the average:", above_average)
