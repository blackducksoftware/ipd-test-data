def filter_even_numbers(input_list):
    even_numbers = []
    even_sum = 0
    
    for num in input_list:
        try:
            if num % 2 == 0:
                even_numbers.append(num)
                even_sum += num
        except TypeError:
            print(f"Error: {num} is not an integer.")
    
    return even_numbers, even_sum

# Example usage
input_list = [1, 2, 3, 4, 5, 6, 'a', 7]
even_numbers, even_sum = filter_even_numbers(input_list)
print("Even numbers in the input list:", even_numbers)
print("Sum of even numbers in the input list:", even_sum)
