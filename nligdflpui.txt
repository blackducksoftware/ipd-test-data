# Function to check if a number is prime
def is_prime(num):
    if num <= 1:
        return False
    if num == 2:
        return True
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            return False
    return True

# Function to generate list of prime numbers up to a specified limit
def generate_prime_list(limit):
    prime_list = []
    num = 2
    while num <= limit:
        if is_prime(num):
            prime_list.append(num)
        num += 1
    return prime_list

# Get user input for limit
limit = int(input("Enter a limit to generate prime numbers up to: "))

# Generate list of prime numbers up to the specified limit
prime_numbers = generate_prime_list(limit)

# Print the list of prime numbers
print("Prime numbers up to", limit, "are:", prime_numbers)
