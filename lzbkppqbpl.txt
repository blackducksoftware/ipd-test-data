def calculate_average(numbers):
    total = sum(numbers)
    average = total / len(numbers)
    return average

def main():
    num_list = [int(x) for x in input("Enter a list of integers separated by a space: ").split()]
    avg = calculate_average(num_list)
    print(f"The average of the numbers in the list is: {avg}")

if __name__ == "__main__":
