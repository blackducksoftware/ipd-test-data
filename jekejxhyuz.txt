```python
def sum_of_even_numbers(numbers):
    total_sum = 0
    try:
        for num in numbers:
            if int(num) % 2 == 0:
                total_sum += int(num)
        
        return total_sum
    except ValueError:
        return "Please enter valid numbers in the list."

# Take input from the user
input_numbers = input("Enter a list of numbers separated by spaces: ").split()

# Check if the input list is empty
if not input_numbers:
    print("List is empty, please enter some numbers.")
else:
    print(f"The sum of all even numbers in the list is: {sum_of_even_numbers(input_numbers)}")
```
