def sum_positive_numbers(nums):
    sum_positive = 0
    for num in nums:
        if num > 0:
            sum_positive += num
    return sum_positive

# Test the function

print(sum_positive_numbers(nums1))  # Output: 6

nums2 = [-1, -2, -3]
print(sum_positive_numbers(nums2))  # Output: 0
