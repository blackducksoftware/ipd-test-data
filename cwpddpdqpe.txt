def is_prime(num):
    if num < 2:
        return False
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            return False
    return True

try:
    max_num = int(input("Enter a maximum number: "))
    if max_num < 2:
        raise ValueError("Please enter a number greater than or equal to 2.")
    
    print("Prime numbers up to", max_num)
    for i in range(2, max_num + 1):
        if is_prime(i):
            print(i)
except ValueError as e:
    print("Error:", e)
