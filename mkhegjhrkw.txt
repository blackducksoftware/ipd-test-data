import sys

def square_even_numbers(input_list):
    try:
        new_list = []
        for num in input_list:
            if num % 2 == 0:
                new_list.append(num**2)
        return new_list
    except Exception as e:
        print(f"An error occurred: {e}")
        return None

if __name__ == "__main__":
    try:
        input_list = list(map(int, input("Enter a list of numbers separated by spaces: ").split()))
        result = square_even_numbers(input_list)
        if result:
            print(f"The new list with even numbers squared is: {result}")
    except ValueError:
        print("Please enter valid numbers.")
    except KeyboardInterrupt:
        print("\nProgram terminated by user.")
        sys.exit(0)
