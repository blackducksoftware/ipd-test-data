```python
# Program to create a list of numbers, calculate sum, average, max value and sort in descending order

# Function to calculate sum of numbers
def calculate_sum(numbers):
    return sum(numbers)

# Function to calculate average of numbers
def calculate_average(numbers):
    return sum(numbers) / len(numbers)

# Function to find maximum value in list
def find_maximum(numbers):
    return max(numbers)

# Take user input to create a list of numbers
numbers = input("Enter the numbers separated by spaces: ")
numbers = list(map(int, numbers.split()))

# Calculate sum, average and max value
total_sum = calculate_sum(numbers)
average = calculate_average(numbers)
maximum = find_maximum(numbers)

# Sort the list in descending order
numbers.sort(reverse=True)

# Print the results
print("Sum of numbers:", total_sum)
print("Average of numbers:", average)
print("Maximum value in the list:", maximum)
print("List sorted in descending order:", numbers)
```
