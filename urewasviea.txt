def sum_even_numbers(numbers):
    total = 0
    for num in numbers:
        if not isinstance(num, int):
            return "Error: Input list contains non-integer values"
        elif num % 2 == 0:
            total += num
    return total

input_list = [2, 3, 6, 8, 10]
print(sum_even_numbers(input_list))  # Output: 26

input_list = [1, 3, 'a', 6, 8, 10]
print(sum_even_numbers(input_list))  # Output: Error: Input list contains non-integer values

input_list = []
print(sum_even_numbers(input_list))  # Output: 0
