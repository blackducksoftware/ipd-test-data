public class SumOfSquaresOfEvens {
    public static int sumOfSquaresOfEvens(int[] numbers) {
        int sumOfSquares = 0;
        for (int num : numbers) {
            if (num % 2 == 0) {
                sumOfSquares += num * num;
            }
        }
        return sumOfSquares;
    }

    public static void main(String[] args) {

        System.out.println(sumOfSquaresOfEvens(numbers)); // Output: 56 (2^2 + 4^2 + 6^2)
    }
}
