# Program to generate the first 20 numbers in the Fibonacci sequence

def fibonacci(n):
    fib_sequence = [0, 1]
    
    for i in range(2, n):
        fib_sequence.append(fib_sequence[-1] + fib_sequence[-2])
    
    return fib_sequence

# Generate the first 20 numbers in the Fibonacci sequence
fibonacci_sequence = fibonacci(20)

# Print the Fibonacci sequence
for num in fibonacci_sequence:
    print(num)
