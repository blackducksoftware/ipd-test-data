def sum_of_squared_even_numbers(numbers):
    try:
        even_numbers = [int(num) for num in numbers if int(num) % 2 == 0]
        squared_even_numbers = [num**2 for num in even_numbers]
        total_sum = sum(squared_even_numbers)
        return total_sum
    except ValueError:
        print("Invalid input. Please enter only numeric values.")
