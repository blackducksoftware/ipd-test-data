import sys

def fibonacci_sequence(n):
    sequence = []
    if n <= 0:
        return sequence
    elif n == 1:
        return [0]
    elif n == 2:
        return [0, 1]
    else:
        sequence = [0, 1]
        for i in range(2, n):
            next_num = sequence[i-1] + sequence[i-2]
            sequence.append(next_num)
        return sequence

try:
    n = int(input("Enter the length of the Fibonacci sequence you want to generate: "))
    if n <= 0:
        print("Please enter a positive number.")
        sys.exit()
    fibonacci = fibonacci_sequence(n)
    print("Fibonacci sequence up to", n, "terms:")
    print(fibonacci)
except ValueError:
    print("Invalid input. Please enter a valid number.")
