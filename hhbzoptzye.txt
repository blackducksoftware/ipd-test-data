import sys

def sum_even_numbers(numbers):
    even_sum = 0
    for num in numbers:
        if num % 2 == 0:
            even_sum += num
    return even_sum

if __name__ == "__main__":
    numbers = []
    if len(sys.argv) > 1:
        numbers = [int(num) for num in sys.argv[1:]]
    
    if not numbers:
        print("No numbers to sum")
    else:
        result = sum_even_numbers(numbers)
        print(f"The sum of all even numbers in the list is: {result}")
