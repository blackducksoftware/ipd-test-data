Here's a simple program in Python that simulates a game of tic-tac-toe:

```python
def print_board(board):
    for row in board:
        print(" | ".join(row))
        print("-" * 5)

def check_winner(board):
    # Check rows
    for row in board:
        if all(mark == row[0] for mark in row):
            return row[0]

    # Check columns
    for col in range(3):
        if all(board[row][col] == board[0][col] for row in range(3)):
            return board[0][col]

    # Check diagonals
    if board[0][0] == board[1][1] == board[2][2] or board[0][2] == board[1][1] == board[2][0]:
        return board[1][1]

    return None

def tic_tac_toe():
    board = [[" " for _ in range(3)] for _ in range(3)]
    player = "X"

    while True:
        print_board(board)

        row = int(input(f"Player {player}, enter row (0-2): "))
        col = int(input(f"Player {player}, enter col (0-2): "))

        if board[row][col] != " ":
            print("That position is already taken. Try again.")
            continue

        board[row][col] = player

        winner = check_winner(board)
        if winner:
            print_board(board)
            print(f"Player {player} wins!")
            break

        if all(all(mark != " " for mark in row) for row in board):
            print_board(board)
            print("It's a draw!")
            break

        player = "O" if player == "X" else "X"

if __name__ == "__main__":
    tic_tac_toe()
```

To play the game, simply run the program and follow the on-screen instructions to input your moves. The program will display the current state of the board after each move and announce the winner or a draw at the end of the game. Enjoy playing tic-tac-toe!
