```python
def find_unique_values(input_list):
    unique_values = list(set(input_list))
    count_unique = len(unique_values)
    
    return unique_values, count_unique

def main():
    input_list = [1, 2, 3, 4, 5, 2, 3, 4, 6, 7, 1, 8, 9, 5]
    
    unique_values, count_unique = find_unique_values(input_list)
    
    print("Original List:")
    print(input_list)
    
    print("\nUnique Values:")
    print(unique_values)
    
    print("\nTotal Number of Unique Values:")
    print(count_unique)

if __name__ == "__main__":
    main()
```
