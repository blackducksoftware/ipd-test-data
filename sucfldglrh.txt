def sum_of_even_numbers(input_list):
    sum_even = 0
    for num in input_list:
        if num % 2 == 0:
            sum_even += num
    if sum_even > 100:
        print("The sum is greater than 100")
    else:
        print("The sum is less than or equal to 100")
    return sum_even

# Taking input from the user
