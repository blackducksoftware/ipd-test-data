def sum_even_numbers(numbers):
    try:
        total = 0
        has_even_numbers = False
        
        for num in numbers:
            if num % 2 == 0:
                total += num
                has_even_numbers = True
        
        if has_even_numbers:
            print("Sum of even numbers in the list:", total)
        else:
            print("There are no even numbers in the list.")
    
    except TypeError:
        print("Invalid input. Please provide a list of numbers.")
    except:
        print("An error occurred. Please try again.")

# Test the program
try:
    numbers = [1, 2, 3, 4, 5, 6]
    sum_even_numbers(numbers)
    
    numbers = [1, 3, 5, 7]
    sum_even_numbers(numbers)
    
    numbers = "invalid input"
    sum_even_numbers(numbers)
except:
    print("An error occurred. Please try again.")
