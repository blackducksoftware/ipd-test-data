def is_prime(num):
    if num < 2:
        return False
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            return False
    return True

def prime_numbers_up_to(n):
    prime_numbers = []
    for i in range(2, n + 1):
        if is_prime(i):
            prime_numbers.append(i)
    return prime_numbers

user_input = int(input("Enter a number: "))
result = prime_numbers_up_to(user_input)
print("Prime numbers up to", user_input, "are:", result)
