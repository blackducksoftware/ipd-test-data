```python
def sum_of_even_numbers(numbers):
    total = 0
    for num in numbers:
        if num % 2 == 0:
            total += num
    return total

input_numbers = [2, 4, 6, 8, 10, 15, 17, 20]
result = sum_of_even_numbers(input_numbers)
print(result)
```
