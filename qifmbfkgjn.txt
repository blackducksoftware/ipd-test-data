```python
def remove_duplicates_and_sort(input_list):
    unique_list = list(set(input_list))  # Remove duplicates by converting the list to a set
    unique_list.sort()  # Sort the list in ascending order
    return unique_list

input_list = [9, 3, 6, 1, 9, 2, 3, 6]
sorted_list = remove_duplicates_and_sort(input_list)

print(sorted_list)
```
