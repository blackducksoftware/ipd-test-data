# Function to calculate and return the sum of even numbers in a list
def sum_of_even_numbers(numbers):
    even_numbers = [num for num in numbers if num % 2 == 0]
    sum_of_even = sum(even_numbers)
    return sum_of_even

# Read a list of integers from the user
numbers = input("Enter a list of integers separated by spaces: ")
numbers = list(map(int, numbers.split()))

# Calculate and print the sum of even numbers in the list
sum_even = sum_of_even_numbers(numbers)
print(f"The sum of even numbers in the list is: {sum_even}")
