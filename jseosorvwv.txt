Here is a Python program that calculates the average of a list of numbers input by the user:

```python
def calculate_average(numbers):
    total = sum(numbers)
    average = total / len(numbers)
    return average

def main():
    number_list = []
    while True:
        try:
            number = float(input("Enter a number (or press enter to finish): "))
            if number == "":
                break
            number_list.append(number)
        except ValueError:
            print("Invalid input. Please enter a valid number.")

    if number_list:
        average = calculate_average(number_list)
        print(f"The average of the numbers is: {average}")
    else:
        print("No numbers were entered.")

if __name__ == "__main__":
    main()
```

You can run this program and input a list of numbers to calculate their average. The program will stop accepting input when you press the enter key without typing a number.
