```python
def input_validation(input_list):
    if not input_list:
        print("List is empty. Please provide a list of numbers.")
        return False
    for num in input_list:
        if not isinstance(num, int):
            print("Invalid input. Please provide a list of numbers.")
            return False
    return True

def sum_even_numbers(input_list):
    total = 0
    for num in input_list:
        if num % 2 == 0:
            total += num
    return total

def main():
    input_list = input("Enter a list of numbers separated by spaces: ").split()
    input_list = [int(num) for num in input_list]
    
    if input_validation(input_list):
        sum_of_evens = sum_even_numbers(input_list)
        print("Sum of even numbers in the list is:", sum_of_evens)

if __name__ == "__main__":
    main()
```

Example Input/Output:
```
Enter a list of numbers separated by spaces: 1 2 3 4 5 6 7 8 9
Sum of even numbers in the list is: 20

Enter a list of numbers separated by spaces: -2 -1 0 1
Sum of even numbers in the list is: -2

Enter a list of numbers separated by spaces: 1 3 5 7
No even numbers in the list.
```
