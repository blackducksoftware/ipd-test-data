```python
def sum_even_and_avg_odd(numbers):
    even_sum = 0
    odd_count = 0
    odd_sum = 0

    if len(numbers) == 0:
        return 0, 0
    else:
        for num in numbers:
            if num % 2 == 0:
                even_sum += num
            else:
                odd_sum += num
                odd_count += 1

        if odd_count == 0:
            odd_avg = 0
        else:
            odd_avg = odd_sum / odd_count

        return even_sum, odd_avg

# Test cases
print(sum_even_and_avg_odd([]))  # Edge case: empty list
print(sum_even_and_avg_odd([2, 4, 6]))  # Edge case: list with only even numbers
print(sum_even_and_avg_odd([1, 3, 5]))  # Edge case: list with only odd numbers
print(sum_even_and_avg_odd([1, 2, 3, 4, 5, 6]))  # Mixed list
```

This program defines a function `sum_even_and_avg_odd` that takes a list of numbers as input, calculates the sum of all even numbers, and calculates the average of all odd numbers in the list.
