def fibonacci(n):
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    elif n == 2:
        return [0, 1]
    else:
        seq = fibonacci(n-1)
        seq.append(seq[-1] + seq[-2])
        return seq

n = int(input("Enter the number of terms in Fibonacci sequence: "))
fib_seq = fibonacci(n)

print("Fibonacci sequence up to", n, "terms:")
print(fib_seq)
