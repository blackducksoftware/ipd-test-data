```python
def sum_even_numbers(numbers):
    sum_even = 0
    for num in numbers:
        if num % 2 == 0:
            sum_even += num
    return sum_even

# Taking input from the user
numbers = input("Enter a list of numbers separated by commas: ")
numbers_list = list(map(int, numbers.split(',')))

# Calling the function to calculate the sum of even numbers in the list
result = sum_even_numbers(numbers_list)

# Displaying the result
print("The sum of all even numbers in the list is:", result)
```
