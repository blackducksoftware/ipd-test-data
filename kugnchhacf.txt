```python
import requests
import tkinter as tk
from tkinter import messagebox

def get_weather():
    city = city_entry.get()
    api_key = 'YOUR_API_KEY_HERE'  # Replace with your API key from OpenWeatherMap
    url = f'http://api.openweathermap.org/data/2.5/weather?q={city}&appid={api_key}&units=metric'
    
    try:
        response = requests.get(url)
        data = response.json()
        temp = data['main']['temp']
        weather = data['weather'][0]['description']
        result_label.config(text=f'Temperature: {temp}Â°C\nWeather: {weather}')
    except:
        messagebox.showerror('Error', 'City not found or API connection issue')

root = tk.Tk()
root.title('Weather App')

city_label = tk.Label(root, text='Enter city:')
city_label.pack()

city_entry = tk.Entry(root)
city_entry.pack()

fetch_btn = tk.Button(root, text='Get Weather', command=get_weather)
fetch_btn.pack()

result_label = tk.Label(root, text='')
result_label.pack()

root.mainloop()
```

Make sure to replace `'YOUR_API_KEY_HERE'` with your actual API key from OpenWeatherMap. You can sign up for a free API key on their website. This app will display the current temperature and weather conditions for the city entered by the user. It will handle errors gracefully if the city is not found or if there is an issue with the API connection.
