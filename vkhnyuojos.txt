Here is a Python code that simulates a blackjack game:

```python
import random

def deal_card():
    cards = [2, 3, 4, 5, 6, 7, 8, 9, 10, 10, 10, 10, 11]
    return random.choice(cards)

def calculate_hand_value(hand):
    total = sum(hand)
    num_aces = hand.count(11)
    
    while total > 21 and num_aces:
        total -= 10
        num_aces -= 1
    
    return total

def check_win_loss(player_total, dealer_total):
    if player_total == 21:
        return "Blackjack! You win!"
    elif player_total > 21:
        return "Bust! You lose!"
    elif dealer_total == 21:
        return "Dealer has Blackjack! You lose!"
    elif dealer_total > 21:
        return "Dealer busts! You win!"
    elif player_total > dealer_total:
        return "You win!"
    elif player_total < dealer_total:
        return "You lose!"
    else:
        return "Push (tie)! Play again!"

def blackjack():
    player_hand = [deal_card(), deal_card()]
    dealer_hand = [deal_card(), deal_card()]
    
    print(f"Player's hand: {player_hand}")
    print(f"Dealer's hand: [X, {dealer_hand[1]}]")
    
    player_total = calculate_hand_value(player_hand)
    dealer_total = calculate_hand_value(dealer_hand)
    
    while player_total < 21:
        action = input("Do you want to hit or stand? ").lower()
        
        if action == "hit":
            player_hand.append(deal_card())
            player_total = calculate_hand_value(player_hand)
            print(f"Player's hand: {player_hand}")
        elif action == "stand":
            break
        else:
            print("Please enter 'hit' or 'stand'.")
    
    while dealer_total < 17:
        dealer_hand.append(deal_card())
        dealer_total = calculate_hand_value(dealer_hand)
    
    print(f"Player's total: {player_total}")
    print(f"Dealer's total: {dealer_total}")
    
    result = check_win_loss(player_total, dealer_total)
    print(result)

blackjack()
```

You can run this code in a Python environment to play the blackjack game. Enjoy!
