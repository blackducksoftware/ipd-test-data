```python
def is_prime(num):
    if num < 2:
        return False
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            return False
    return True

def generate_prime_list(limit):
    prime_list = []
    for i in range(2, limit):
        if is_prime(i):
            prime_list.append(i)
    return prime_list

def calculate_sum_of_primes(prime_list):
    return sum(prime_list)

# Get user input for the limit
limit = int(input("Enter a number to generate prime numbers up to: "))

# Generate the list of prime numbers
prime_list = generate_prime_list(limit)

# Calculate the sum of prime numbers
sum_of_primes = calculate_sum_of_primes(prime_list)

# Print the prime numbers and their sum
print("List of prime numbers up to", limit, ":")
print(prime_list)
print("Sum of prime numbers:", sum_of_primes)
```
